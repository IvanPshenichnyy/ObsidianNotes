```bsl
// Процедура - Проверить подразделение
//
// Параметры:
//  Форма - ФормаКлиентскогоПриложения - форма в которой выполняется вызов этой функции.
//  ОтменитьОбработку - Булево - Отменить выполнение операции
//
Асинх Процедура ПроверитьПодразделение(Форма, ЕстьВопросыПередЗаписью)
    
    ТекущийОбъект = Форма.Объект;
    Если ТекущийОбъект.Свойство("Подразделение") = Ложь Тогда
        Возврат;
    КонецЕсли;
    
    Если Не ЗначениеЗаполнено(ТекущийОбъект.Подразделение) Тогда
        ОбщегоНазначенияКлиентСервер.СообщитьПользователю("Заполните реквизит подразделение.",
        ,
        "Подразделение",
        "Объект");
        ЕстьВопросыПередЗаписью = Истина;
        Возврат;
    КонецЕсли;
    
    Если ТекущийОбъект.Свойство("Склад") = Истина Тогда
        ПодразделениеСклад = ТекущийОбъект.Склад.Подразделение;
        Если ТекущийОбъект.Подразделение <> ПодразделениеСклад Тогда
            Ответ = Ждать ВопросАсинх(СтрШаблон("Подразделение в ""Документе"" %1 
            |Отличается от подразделения ""Склада"" %2
            |Продолжить операцию?",
            ТекущийОбъект.Подразделение,
            ПодразделениеСклад),
            РежимДиалогаВопрос.ДаНет, 60);    
            Если Ответ = КодВозвратаДиалога.Да Тогда
                ЕстьВопросыПередЗаписью = Истина;
            КонецЕсли;
        КонецЕсли;
    КонецЕсли; 
    
    Возврат;    
    
КонецПроцедуры
```
